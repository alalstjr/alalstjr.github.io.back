# 확장형 포문 정리요망

## 2차원 배열 출력
	public static void main(String[] args) {
		int point = 0;
		int[][] omok = new int[10][10];
		
		for(int[] i : omok) {
			System.out.print("\n");
			for(int j : i) {
				point++;
				System.out.print(point + "\t");	
			}
		}
	}
